## Context

The previous iteration failed to compile due to mismatched feature repository names and unexposed cache clearing contracts. To fix this without coupling the administrative reset tool to every feature package, we refactor the use case to clear the database dynamically using Room 3's connection pool, mirroring the schemaless design used in our viewer.

## Design

1. **Dynamic Table Wiping.** Instead of invoking individual repositories, we look up user tables dynamically from the system catalog and execute targeted `DELETE FROM` operations within a single write transaction connection.
2. **Framework Decoupling.** This entirely eliminates the compilation dependency on `CatRepository`, `CoinRepository`, and `CarRepository` import headers.

## Verification gate

1. `./gradlew assembleDebug`
2. Expect: Compiles clean; domain data-clearing mechanics resolve flawlessly.

package com.icodeforyou.catscarscoins.dbviewer.domain

import androidx.room3.RoomDatabase
import androidx.room3.useWriterConnection
import kotlinx.coroutines.Dispatchers
import kotlinx.coroutines.withContext

/**
 * Domain command orchestrator to wipe application footprints across all storage spaces.
 * Drops database table contents dynamically via low-level write connections.
 */
class ResetAppUseCase(
    private val database: RoomDatabase
) {

    /**
     * Executes sequential, full-footprint purges across underlying local engines.
     */
    suspend fun execute(): Unit = withContext(Dispatchers.IO) {
        val findTablesSql = """
            SELECT name FROM sqlite_master 
            WHERE type='table' 
              AND name NOT LIKE 'sqlite_%' 
              AND name NOT LIKE 'room_%'
              AND name NOT LIKE '%_fts%'
              AND name NOT LIKE '%_percent%'
              AND name NOT LIKE '%_segdir%'
              AND name NOT LIKE '%_segments%'
              AND name NOT LIKE '%_docsize%'
              AND name NOT LIKE '%_stat%'
        """.trimIndent()

        try {
            database.useWriterConnection { connection ->
                // Gather target tables dynamically
                val tables = connection.usePrepared(findTablesSql) { statement ->
                    val list = mutableListOf<String>()
                    while (statement.step()) {
                        list.add(statement.getText(0))
                    }
                    list
                }

                // Wipe table records sequentially within the connection scope
                tables.forEach { tableName ->
                    val sanitized = tableName.replace(Regex("[^a-zA-Z0-9_]"), "")
                    if (sanitized.isNotEmpty()) {
                        connection.usePrepared("DELETE FROM $sanitized") { statement ->
                            statement.step()
                        }
                    }
                }
            }
        } catch (e: Exception) {
            // Sunk exception to maintain clean execution flow
        }
    }
}
