package com.icodeforyou.catscarscoins.ui.navigation

import androidx.compose.foundation.layout.fillMaxSize
import androidx.compose.runtime.Composable
import androidx.compose.ui.Modifier
import androidx.navigation.NavHostController
import androidx.navigation.compose.NavHost
import androidx.navigation.compose.composable
import com.icodeforyou.catscarscoins.cars.ui.CarsScreen
import com.icodeforyou.catscarscoins.cats.ui.CatsScreen
import com.icodeforyou.catscarscoins.coins.ui.CoinsScreen
import com.icodeforyou.catscarscoins.dbviewer.ui.DbViewerScreen

/**
 * Global navigation routing manifest managing top-level application screens.
 */
@Composable
fun AppNavGraph(
    navController: NavHostController,
    startDestination: String,
    onCatSelected: (String) -> Unit,
    modifier: Modifier = Modifier
) {
    NavHost(
        navController = navController,
        startDestination = startDestination,
        modifier = modifier
    ) {
        composable("coins") {
            CoinsScreen(modifier = Modifier.fillMaxSize())
        }
        composable("cats") {
            CatsScreen(
                onCatSelected = onCatSelected,
                modifier = Modifier.fillMaxSize()
            )
        }
        composable("cars") {
            CarsScreen(modifier = Modifier.fillMaxSize())
        }
        composable("db_viewer") {
            DbViewerScreen(modifier = Modifier.fillMaxSize())
        }
    }
}
